1function [v, omega] = lidar2force_field(lidarScan, state, v_measured, omega_measured, dt)
 2
 3%% Define here parameters
 4point_closeness                      = 0.15; % distance between two point belonging to the same obstacle
 5walker_positions                     = {[0;0]}; % positions around which the fields is computed. Use more than one position to filter
 6% Grad_U                               = @(R, U_0_alpha_B, r_alpha_B_x, r_alpha_B_y)[-(U_0_alpha_B.*exp(-sqrt(abs(r_alpha_B_x).^2+abs(r_alpha_B_y).^2)./R).*abs(r_alpha_B_x).*sign(r_alpha_B_x).*1.0./sqrt(abs(r_alpha_B_x).^2+abs(r_alpha_B_y).^2))./R;-(U_0_alpha_B.*exp(-sqrt(abs(r_alpha_B_x).^2+abs(r_alpha_B_y).^2)./R).*abs(r_alpha_B_y).*sign(r_alpha_B_y).*1.0./sqrt(abs(r_alpha_B_x).^2+abs(r_alpha_B_y).^2))./R];
 7[R_potential, U_0_alpha_B_potential] = tune_potential(0.5, 1, 10, 0.01);
 8[R_vortex, U_0_alpha_B_vortex]       = tune_potential(0.5, 3, 3, 1);
 9max_deceleration                     = 10;
10braking_angle                        = 75 * pi/180;
11k_prj_omega                          = 2;
12omega_max                            = 45 * pi /180;
13v_des                                = 1;
14v_max                                = 1;
15k_i_vel                              = 1;
16%% Generate angles from lidar scan
17angles = linspace(lidarScan.angle_min,lidarScan.angle_max,numel(lidarScan.ranges));
18
19
20%% Elaborate lidar data
21% put the obstacles measured in groups
22obst_group = group_obstacles(angles, lidarScan.ranges, point_closeness);
23
24
25%% Filtered Force fields
26% initialize vortex and
27repulsive_force  = [0;0];
28vortex_force     = [0;0];
29
30for kk = 1:length(walker_positions)
31    points         = point_obst(obst_group, angles, lidarScan.ranges, walker_positions{kk}); % find the closest point for each obstacle
32    %% Repulsive actions
33    tmp             = size(points);
34    for j = 1:tmp(2) % number of column of points
35        r_alpha         = [0;0];
36        r_B             = [points(1,j); points(2,j)];
37        r_alpha_B       = r_alpha - r_B;
38        r_alpha_B_x     = r_alpha_B(1);
39        r_alpha_B_y     = r_alpha_B(2);
40        repulsive_force = repulsive_force - Grad_U(R_potential, U_0_alpha_B_potential,r_alpha_B_x, r_alpha_B_y);
41    end
42    %% Vortex force
43    for j = 1:tmp(2) % number of column of points
44        r_alpha         = [0;0];
45        r_B             = [points(1,j); points(2,j)];
46        r_alpha_B       = r_alpha - r_B;
47        r_alpha_B_x     = r_alpha_B(1);
48        r_alpha_B_y     = r_alpha_B(2);
49        tmp             = Grad_U(R_vortex, U_0_alpha_B_vortex, r_alpha_B_x, r_alpha_B_y);
50        grad_x          = tmp(1);
51        grad_y          = tmp(2);
52        positive_vortex = [grad_y; -grad_x];
53        
54        
55        vortex_dot_robot = dot([1;0], positive_vortex);
56        if vortex_dot_robot < 0
57            positive_vortex = -positive_vortex;
58        end
59        vortex_force = vortex_force + positive_vortex;
60    end
61    
62end
63vortex_force     = vortex_force    / length(walker_positions);
64repulsive_force  = repulsive_force / length(walker_positions);
65
66%% Total force
67total_force  = repulsive_force + vortex_force * 1;
68
69
70%% Run controller
71%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
72% choose the forward vel
73%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
74
75force_angle   = dot([1;0], total_force / (eps + norm(total_force)));   % cos(angle) between force and vehicle
76if force_angle >= cos(braking_angle) % then acelerate using a PI to the desired speed
77    v = v_measured + dt * k_i_vel * (v_des - v_measured);
78    v = min(v, v_max); % this should be useless
79    v = max(0, v);
80else % then brake
81    braking_action = max_deceleration *  (force_angle - cos(braking_angle));
82    v              = v_measured + dt * braking_action;
83    v              = max(0, v);
84end
85
86%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
87% choose the angular vel
88%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
89omega = k_prj_omega * dot([0;1], total_force);
90omega = min(omega_max, max(-omega_max, omega));
